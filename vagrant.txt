# -*- mode: ruby -*-
# vi: set ft=ruby :

Vagrant.configure("2") do |config|
  config.vm.box = "ubuntu/jammy64"
  config.vm.network "private_network", ip:192.168.169.10
  config.vm.hostname="DockerWordpress"
  config.vm.provider "virtualbox" do |vbox|
     vbox.memory = 4096
     vbox.cpus = 2
     vbox.name ="vm_wordpress"
  end
  config.vm.provision "shell", inline: <<-SHELL
      #Configuration du référentiel apt de Docker.
		sudo apt-get update
		sudo apt-get install -y ca-certificates curl gnupg
		sudo install -m 0755 -d /etc/apt/keyrings
		curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg
		sudo chmod a+r /etc/apt/keyrings/docker.gpg
		echo \
		"deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu \
		$(. /etc/os-release && echo "$VERSION_CODENAME") stable" | \
		sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
		sudo apt-get update
		#installation de Docker-ce 
		sudo apt-get install -y docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin
		#installation de Docker-compose
		
       sudo curl -L "https://github.com/docker/compose/releases/download/1.26.0/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
	   sudo chmod +x /usr/local/bin/docker-compose
	   
	   #installation de l'autocompletion pour les cmds docker et docker compose
	    sudo curl -L https://raw.githubusercontent.com/docker/compose/1.26.0/contrib/completion/bash/docker-compose -o /etc/bash_completion.d/docker-compose
		
	  # lancement du service Docker
		sudo systemctl start docker
		
	  # lancement du service Docker au demarrage de la machine
	    sudo systemctl enable docker
  SHELL
		    
end
